/**
 * @fileoverview Router for order-related operations.
 * Implements routes for managing orders including approving, rejecting,
 * making transactions, and CRUD operations.
 */

const express = require('express');
const orderController = require('../controllers/orderController');
const authController = require('../controllers/authController');

// Create a new route for order operations:
const router = express.Router();

/**
 * Route to approve an order.
 * @name patch/approve/:id
 * @function
 * @memberof module:router
 * @inner
 * @param {string} path - Express path
 * @param {function} middleware - Authentication middleware
 * @param {function} callback - Controller function to approve an order
 */
router.patch(
  '/approve/:id',
  authController.protect,
  orderController.approveOrder,
);

/**
 * Route to reject an order.
 * @name patch/reject/:id
 * @function
 * @memberof module:router
 * @inner
 * @param {string} path - Express path
 * @param {function} middleware - Authentication middleware
 * @param {function} callback - Controller function to reject an order
 */
router.patch(
  '/reject/:id',
  authController.protect,
  orderController.rejectOrder,
);

/**
 * Route to make a transaction on an order.
 * @name patch/transact/:id
 * @function
 * @memberof module:router
 * @inner
 * @param {string} path - Express path
 * @param {function} middleware - Authentication middleware
 * @param {function} callback - Controller function to make a transaction
 */
router.patch(
  '/transact/:id',
  authController.protect,
  orderController.makeTransaction,
);

/**
 * Route to retrieve all orders. Access is restricted to admin users.
 * @name get/
 * @function
 * @memberof module:router
 * @inner
 * @param {string} path - Express path
 * @param {function[]} middleware - Array of authentication and authorization middlewares
 * @param {function} callback - Controller function to get all orders
 */

/**
 * Routes to get, update, or delete a specific order. Access is restricted to admin users.
 * @name route/:id
 * @function
 * @memberof module:router
 * @inner
 * @param {string} path - Express path
 * @param {function[]} middleware - Array of authentication and authorization middlewares
 * @param {function} callback - Controller function to get, update, or delete an order
 */
router
  .route('/')
  .get(
    authController.protect,
    authController.restrictToAdmin,
    orderController.getAllOrders,
  );
router
  .route('/:id')
  .get(
    authController.protect,
    authController.restrictToAdmin,
    orderController.getOrder,
  )
  .patch(
    authController.protect,
    authController.restrictToAdmin,
    orderController.updateOrder,
  )
  .delete(
    authController.protect,
    authController.restrictToAdmin,
    orderController.deleteOrder,
  );

module.exports = router;
